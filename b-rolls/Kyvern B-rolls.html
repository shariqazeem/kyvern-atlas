async function run() {
  // gate every paid call through vault.pay() — chain decides
  const gate = await vault.pay({ merchant: "api.commonstack.ai", amount: 0.02, memo: "chat-completion" });
  if (gate.decision !== "allowed") throw new Error(gate.reason);
